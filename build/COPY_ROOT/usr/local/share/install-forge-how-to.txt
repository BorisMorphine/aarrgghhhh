FORGE INSTALL HOW-TO  
cd /opt/
git clone https://github.com/lllyasviel/stable-diffusion-webui-forge.git 
#Step2
git remote add forge https://github.com/lllyasviel/stable-diffusion-webui-forge.git
git fetch forge
git checkout -b forge-main forge/main
git branch -u forge/main
git pull 
git config pull.rebase false

#Step 3
sudo git init tmp
git config --global init.AUTOMATIC1111/stable-diffusion-webui.git/Master
git remote add upstream main https://github.com/AUTOMATIC1111/stable-diffusion-webui.git
git remote add forked lllyasviel/main https://github.com/lllyasviel/stable-diffusion-webui-forge.git
git fetch upstream and get fetch forked AUTOMATIC1111/MASTER https://github.com/AUTOMATIC1111/stable-diffusion-webui.git
git checkout -m AUTOMATIC1111/Master lllyasviel/main
git merge stable-diffusion-webui-forge AUTOMATIC1111/Master

#Step4
cd /opt/stable-diffusion-webui/stable-diffusion-webui-forge/
sudo apt install python3.10-venv
set=“SD_FORGE”=“/opt/stable-diffusion-webui/extensions/stable-diffusion-webui-forge/“
export LAUNCH_SCRIPT=“launch.py”
python_cmd="python3": “python3 ${INSTALL_DIR}/$(LAUCH_SCRIPT}”

#Step5
cd ${INSTALL_DIR}
micromamba deactivate activate webui
./webui.sh
